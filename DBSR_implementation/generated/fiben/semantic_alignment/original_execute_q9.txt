def execute_q9(db, cfg, param, buy_type_values, sell_type_values):
    listed_security_id = param
    t0 = time.perf_counter()

    txs = list(normalized_collection(db, "Transaction").find({
        "REFERSTO": listed_security_id
    }))

    account_types = defaultdict(set)
    for t in txs:
        account = t.get("ISFACILITATEDBY")
        ttype = t.get("HASTYPE")
        if ttype in buy_type_values:
            account_types[account].add("buy")
        elif ttype in sell_type_values:
            account_types[account].add("sell")

    accounts = [
        acc for acc, types in account_types.items()
        if "buy" in types and "sell" in types
    ]

    latency_ms = (time.perf_counter() - t0) * 1000
    return {
        "latency_ms": latency_ms,
        "success": True,
        "documents_returned": len(accounts),
        "documents_written": 0,
    }
