                doc["_candidate_id"] = candidate_id
                doc["_design_pattern"] = candidate_spec.get("design_pattern")

            inserted = safe_insert_many(coll, docs)
            total += inserted
            _log_collection_progress(collection_name, batch_no, len(docs), total)

    root_pk = ENTITY_PK_MAP.get(root_entity)
    if root_pk:
        try:
            coll.create_index(root_pk)
        except Exception:
            pass

    try:
        coll.create_index("_candidate_id")
        coll.create_index("_benchmark_root_id")
    except Exception:
        pass

    log(
        f"[candidate-load] completed candidate={candidate_id} "
        f"collection={collection_name} total_inserted={total}"
    )

    return {
        "candidate_id": candidate_id,
        "collection_name": collection_name,
        "root_entity": root_entity,
        "documents_inserted": total,
    }
