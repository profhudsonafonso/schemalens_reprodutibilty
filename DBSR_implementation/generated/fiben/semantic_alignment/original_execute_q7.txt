def execute_q7(db, cfg, param, buy_type_values, sell_type_values):
    corporation_id = param
    t0 = time.perf_counter()

    securities = list(normalized_collection(db, "Security").find({"ISPROVIDEDBY": corporation_id}))

    # Correct FIBEN physical join:
    # SECURITIESTRANSACTION.REFERSTO references SECURITY.SECURITYID.
    security_ids = [
        normalize_join_key(s.get("SECURITYID"))
        for s in securities
        if s.get("SECURITYID") is not None
    ]

    transactions = list(normalized_collection(db, "Transaction").find({
        "_norm_REFERSTO": {"$in": security_ids}
    })) if security_ids else []

    by_account = defaultdict(lambda: {"buy": 0.0, "sell": 0.0})
    for t in transactions:
        account = t.get("ISFACILITATEDBY")
        count = float(t.get("HASCOUNT") or 0)
        ttype = t.get("HASTYPE")
        if ttype in buy_type_values:
            by_account[account]["buy"] += count
        elif ttype in sell_type_values:
            by_account[account]["sell"] += count

    winners = [acc for acc, vals in by_account.items() if vals["buy"] > vals["sell"]]

    latency_ms = (time.perf_counter() - t0) * 1000
    return {
        "latency_ms": latency_ms,
        "success": True,
        "documents_returned": len(winners),
        "documents_written": 0,
    }
